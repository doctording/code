import java.io.BufferedReader;

class UnionFind{
    int n;
    private int[] fa;
    private int[] rank;

    public UnionFind(int n){
        this.n = n;
        fa = new int[n+1];
        rank = new int[n+1];
        for(int i=0;i<=n;i++){
            fa[i] = i;
            rank[i] = 1;
        }
    }

    private int find(int a){
        // 递归路径压缩
        if(a == fa[a]){
            return a;
        }
        return fa[a] = find(fa[a]);

        // 并查集路径压缩

    }

    private void unionOperate(int a, int b){
        int finda = find(a);
        int findb = find(b);
        if(finda < findb){
            fa[findb] = finda;
        }else {
            fa[finda] = findb;
        }
    }

    private void unionOperateRank(int a, int b){
        int finda = find(a);
        int findb = find(b);
        if(finda == findb){
            return;
        }
        // 层数大的作为新根
        if(rank[finda] < rank[findb]){
            fa[finda] = findb;
        }else if(rank[finda] > rank[findb]){
            fa[findb] = finda;
        }else {
            // 随意选择一个, 让 findb 成为新根
            fa[finda] = findb;
            rank[findb] = rank[findb] + 1;
        }
    }


    public boolean isSameSet(int a, int b){
        if(find(a) == find(b)){
            return true;
        }
        return false;
    }

    public void union(int a, int b){
        unionOperateRank(a, b);
    }
}
