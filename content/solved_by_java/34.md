# ac(小根堆求解)

```java
import java.util.PriorityQueue;
import java.util.Scanner;


public class Main {

    public static Scanner sc = new Scanner(System.in);

    public static void main(String[] args) {
        int n = sc.nextInt();
        int k = sc.nextInt();

        // 小根堆
        PriorityQueue<Integer> heap = new PriorityQueue<>((a,b)->{
           if(a < b){
               return -1;
           }else if(a > b){
               return 1;
           }
           return 0;
        });

        for(int i=0;i<n;i++){
            int len = sc.nextInt();
            for(int j=0;j<len;j++){
                int var = sc.nextInt();
                if(heap.size() < k){
                    heap.add(var);
                }else if(heap.size() == k){
                    if(var > heap.peek()){
                        heap.poll();
                        heap.add(var);
                    }
                }
            }
        }
        int[] ans = new int[k];
        int ansIndex = k-1;
        while(!heap.isEmpty()){
            int var = heap.poll();
            ans[ansIndex --] = var;
        }
        for(int i=0;i<k-1;i++) {
            System.out.print(ans[i] + " ");
        }
        System.out.println(ans[k-1]);
    }
}
```
