# ac

时间复杂度：O(n^2), 剪枝

```java

import java.util.Scanner;

public class Main {

    public static Scanner sc = new Scanner(System.in);

    /*
5 0
1 -2 1 1 1

sums
0 1 -1 0 1 2
     */
    public static void main(String[] args) {

        int n = sc.nextInt();
        int k = sc.nextInt();
        int[] arr = new int[n];
        // sum[j] - sums[i] : 区间：[i, j-1]
        // sum[j + 1] - sum[i] : 区间：[i, j]
        int[] sums = new int[n + 1];
        for(int i=0;i<n;i++){
            arr[i] = sc.nextInt();
            if(i == 0){
                sums[i + 1] = arr[i];
            }else {
                sums[i + 1] = sums[i] + arr[i];
            }
        }
        int ans = 0;
        for(int i=0;i<n;i++){
            for(int j=i;j<n;j++){
                int tmp;
                int len = j - i + 1;
                if(len <= ans){
                    continue;
                }
                if(j == i){
                    tmp = arr[i];
                }else {
                    tmp = sums[j + 1] - sums[i];
                }
                if(tmp == k){
                    ans = Math.max(ans, len);
                }
            }
            if(n - i < ans){
                break;
            }
        }
        System.out.println(ans);
    }

}
```
